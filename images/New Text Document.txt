 <!-- Total price section -->
        <section class="ml-12 mt-4 text-[#111] text-xl font-medium">
            <p>Total Price: <span id="totalPrice">0.00 TK</span></p>
        </section>

                <script>
                       let totalPrice = 0;

function calculateTotalPrice() {
    // Calculate the total price of selected items
    const selectedItems = document.querySelectorAll('input[type="checkbox"]:checked');
    totalPrice = 0;

    selectedItems.forEach((item) => {
        totalPrice += parseFloat(item.dataset.price);
    });

    // Update the total price display
    document.getElementById('totalPrice').textContent = totalPrice.toFixed(2) + ' TK';
}

function applyCoupon() {
    const couponCode = document.getElementById('couponCode').value;
    if (couponCode === 'SELL200' && totalPrice >= 200) {
        // Apply a 20% discount
        totalPrice *= 0.8;
        document.getElementById('totalPrice').textContent = totalPrice.toFixed(2) + ' TK';
        alert('Coupon applied successfully!');
    } else {
        alert('Invalid coupon code or total price is less than 200 TK.');
    }
}

// Add event listeners to checkboxes
const checkboxes = document.querySelectorAll('input[type="checkbox"]');
checkboxes.forEach((checkbox) => {
    checkbox.addEventListener('change', calculateTotalPrice);
});
                </script>
